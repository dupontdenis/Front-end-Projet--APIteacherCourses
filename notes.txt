// GET /teachers?ids=691c6...,691d...
// Server (Express + Mongoose)
router.get('/teachers', async (req, res) => {
  const ids = (req.query.ids || '').split(',').filter(Boolean);
  const teachers = await Teacher.find({ _id: { $in: ids } }, 'name'); // only name field
  res.json(teachers);
});
